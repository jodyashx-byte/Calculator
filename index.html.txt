<script>
// SW
if ('serviceWorker' in navigator) {
  navigator.serviceWorker.register('sw.js');

// TAB
function switchTab(tab, el){
  document.querySelectorAll('.tab').forEach(t=>t.classList.remove('active'));
  el.classList.add('active');

  document.querySelectorAll('.section').forEach(s=>s.classList.remove('activeSection'));
  document.getElementById(tab).classList.add('activeSection');
}

// CALC ENGINE
let display=document.getElementById("displayText");
let current="";

function update(){
  display.innerText=current||"0";
}

function number(n){
  current+=n;
  update();
}

function operator(op){
  if(current==="") return;
  let last=current.slice(-1);
  if(['+','-','*','/'].includes(last)){
    current=current.slice(0,-1)+op;
  } else {
    current+=op;
  }
  update();
}

function decimal(){
  let parts=current.split(/[\+\-\*\/]/);
  let last=parts[parts.length-1];
  if(!last.includes(".")){
    current+=".";
    update();
  }
}

function clearDisplay(){
  current="";
  update();
}

function plusMinus(){
  if(!current) return;
  current=(parseFloat(current)*-1).toString();
  update();
}

function percent(){
  if(!current) return;
  current=(parseFloat(current)/100).toString();
  update();
}

function calculate(){
  try{
    current=Function('"use strict";return('+current+')')().toString();
  }catch{
    current="";
    display.innerText="Error";
  }
  update();
}

// CONVERTER
function convert(){
  let v=parseFloat(val.value);
  if(isNaN(v)) return;

  let f=from.value;
  let t=to.value;
  let r=v;

  if(f=="kg"&&t=="g") r=v*1000;
  else if(f=="g"&&t=="kg") r=v/1000;

  else if(f=="kb"&&t=="mb") r=v/1024;
  else if(f=="mb"&&t=="kb") r=v*1024;
  else if(f=="mb"&&t=="gb") r=v/1024;
  else if(f=="gb"&&t=="mb") r=v*1024;

  result.innerText=r;
}

// PERCENT
function calcPercent(){
  let g=parseFloat(got.value);
  let t=parseFloat(total.value);

  if(isNaN(g)||isNaN(t)||t===0){
    percentResult.innerText="Invalid";
    return;
  }

  let p=(g/t)*100;
  percentResult.innerText=p.toFixed(2)+" %";
}
